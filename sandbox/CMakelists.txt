cmake_minimum_required(VERSION 3.28)
project(Application LANGUAGES CXX)

# Set the C++ standard to C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_CXX_EXTENSIONS OFF)

# Use the multi-threaded DLL runtime library
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreadedDLL")


# Add the main executable
add_executable(application
  src/sandboxApp.cpp
)

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/Mir/src)

# Define macros
target_compile_definitions(application PRIVATE MIR_PLATFORM_WINDOWS)
# Link the engine DLL
target_link_libraries(application PUBLIC Mir INTERFACE imgui)

# Set the output directory for the executable
set_target_properties(application PROPERTIES
  RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
